309  if (bounds.start < bounds.min)

311  else if (bounds.start > bounds.max)
312  bounds.max = bounds.start;

318  Y& y = pt.second.first;
319  y = ossia::easing::ease{}(bounds.min, bounds.max, y);
320  }
321  }
322  else if (bounds.start == bounds.end)

333  return bounds.max - val * (bounds.max - bounds.min);
