43  constexpr void loop(ossia::time_value start_offset, ossia::time_value loop_duration, Exec f, Transport transport) const noexcept
44  {
45  ossia::token_request other = *this;
46  ossia::time_value orig_from = other.prev_date;
47  ossia::time_value tick_amount = other.date - other.prev_date;
48 
49  while (tick_amount > 0_tv)
50  {
51  const time_value cur_from{orig_from % loop_duration};
52  if (cur_from + tick_amount < loop_duration)
53  {
54  other.prev_date = cur_from + start_offset;
55  other.date = other.prev_date + tick_amount;
56  f(other);
57  break;
58  }
59  else
60  {
61  auto this_tick = loop_duration - cur_from;
62 
63  tick_amount -= this_tick;
64  orig_from += this_tick;
65  other.prev_date = cur_from + start_offset;
66  other.date = other.prev_date + this_tick;
67 
68  f(other);
69 
70  transport(start_offset);
71  other.offset += this_tick;
72  }
73  }
74  }

179  constexpr std::optional<time_value> get_quantification_date(double rate) const noexcept
180  {
181  std::optional<time_value> quantification_date;
182 
183  if(rate <= 0.)
184  return prev_date;
185 
186  const double musical_tick_duration = musical_end_position - musical_start_position;
187  if(musical_tick_duration <= 0.)
188  return prev_date;
189 
190  if(rate <= 1.)
191  {
192  // Quantize relative to bars
193  if(musical_end_last_bar != musical_start_last_bar)
194  {
195  // There is a bar change in this tick
196  const double musical_bar_start = musical_end_last_bar - musical_start_position;
197 
198  const double ratio = musical_bar_start / musical_tick_duration;
199  const time_value dt = date - prev_date; // TODO should be tick_offset
200 
201  quantification_date = prev_date + dt * ratio;
202  }
203  }
204  else
205  {
206  // Quantize relative to quarter divisions
207  // TODO ! if there is a bar change,
208  // and no prior quantization date before that, we have to quantize to the bar change
209  const double start_quarter = (musical_start_position - musical_start_last_bar);
210  const double end_quarter = (musical_end_position - musical_start_last_bar);
211 
212  // duration of what we quantify in terms of quarters
213  const double musical_quant_dur = rate / 4.;
214  const double start_quant = std::floor(start_quarter * musical_quant_dur);
215  const double end_quant = std::floor(end_quarter * musical_quant_dur);
216 
217  if(start_quant != end_quant)
218  {
219  // Date to quantify is the next one :
220  const double musical_tick_duration = musical_end_position - musical_start_position;
221  const double quantified_duration = (musical_start_last_bar + (start_quant + 1) * 4. / rate) - musical_start_position;
222  const double ratio = (date - prev_date).impl / musical_tick_duration;
223 
224  quantification_date = prev_date + quantified_duration * ratio;
225  }
226  }
227 
228  return quantification_date;
229  }

239  template<typename Tick, typename Tock>
240  constexpr void metronome(double modelToSamplesRatio, Tick tick, Tock tock) const noexcept
241  {
242  if((musical_end_last_bar != musical_start_last_bar) || musical_start_position == 0.)
243  {
244  // There is a bar change in this tick, start the up tick
245  const double musical_tick_duration = musical_end_position - musical_start_position;
246  const double musical_bar_start = musical_end_last_bar - musical_start_position;
247  const int64_t samples_tick_duration = physical_write_duration(modelToSamplesRatio);
248  if(samples_tick_duration > 0)
249  {
250  const double ratio = musical_bar_start / musical_tick_duration;
251  const int64_t hi_start_sample = samples_tick_duration * ratio;
252  tick(hi_start_sample);
253  }
254  }
255  else
256  {
257  const int64_t start_quarter = std::floor(musical_start_position - musical_start_last_bar);
258  const int64_t end_quarter = std::floor(musical_end_position - musical_start_last_bar);
259  if(start_quarter != end_quarter)
260  {
261  // There is a quarter change in this tick, start the down tick
262  // start_position is prev_date
263  // end_position is date
264  const double musical_tick_duration = musical_end_position - musical_start_position;
265  const double musical_bar_start = (end_quarter + musical_start_last_bar) - musical_start_position;
266  const int64_t samples_tick_duration = physical_write_duration(modelToSamplesRatio);
267  if(samples_tick_duration > 0)
268  {
269  const double ratio = musical_bar_start / musical_tick_duration;
270  const int64_t lo_start_sample = samples_tick_duration * ratio;
271  tock(lo_start_sample);
272  }
273  }
274  }

277  constexpr bool unexpected_bar_change() const noexcept
278  {
279  double bar_difference = musical_end_last_bar - musical_start_last_bar;
280  if(bar_difference != 0.)
281  {
282  // If the difference is divisble by the signature,
283  // then the bar change is expected.
284  // e.g. start = 4 -> end = 8 ; signature = 4/4 : good
285  // e.g. start = 4 -> end = 8 ; signature = 6/8 : bad
286  // e.g. start = 4 -> end = 7 ; signature = 6/8 : good
287 
288  double quarters_sig = 4. * double(signature.upper) / signature.lower;
289  double div = bar_difference / quarters_sig;
290  bool unexpected = div - int64_t(div) > 0.000001;
291  return unexpected;
292  }
293  return false;
294  }
